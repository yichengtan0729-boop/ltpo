keep = 0
with open(src, 'r', encoding='utf-8') as fin, open(dst, 'w', encoding='utf-8') as fout:
    for line in fin:
        line = line.strip()
        if not line:
            continue
        obj = json.loads(line)
        if float(obj.get('reliability', 0.0)) >= 0.89:
            fout.write(json.dumps(obj, ensure_ascii=False) + '\n')
            keep += 1
print('kept:', keep)
print('saved to:', dst)
